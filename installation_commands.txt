# ChemFinder Installation Commands
# Copy and run these commands in order

# 1. Create new project
npm create vite@latest chemfinder -- --template react

# 2. Navigate to project folder
cd chemfinder

# 3. Install all dependencies in one command
npm install @headlessui/react @heroicons/react framer-motion @tailwindcss/forms react-router-dom papaparse react@18.2.0 react-dom@18.2.0

# 4. Install development dependencies
npm install -D tailwindcss postcss autoprefixer @vitejs/plugin-react@4.2.1 eslint eslint-plugin-react eslint-plugin-react-hooks eslint-plugin-react-refresh

# 5. Initialize Tailwind CSS
npx tailwindcss init -p

# 6. Create necessary folders
mkdir src\components
mkdir public\data

# 7. Copy your files
# - Copy all .jsx files to src/components/
# - Copy your CSV file to public/data/Supplier_Ingredient_Data.csv

# 8. Start the development server
npm run dev

# Note: Make sure Node.js (v14+) is installed on your system before running these commands

# Troubleshooting Commands:
# If you encounter any issues, run these commands:

# Clean installation:
rm -rf node_modules
rm package-lock.json
npm install

# Update all dependencies:
npm update

# Check for errors:
npm audit fix

# For Windows PowerShell users:
# Replace 'rm -rf' with 'Remove-Item -Recurse -Force'
# Example:
# Remove-Item -Recurse -Force node_modules
# Remove-Item package-lock.json

# Required Environment:
# - Node.js: v14 or higher
# - npm: v6 or higher
